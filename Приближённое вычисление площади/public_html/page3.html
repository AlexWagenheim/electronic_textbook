Задача

Составить и отладить программу определения площади представленной ниже фигуры методом Монте-Карло.

 

 

Решение (Java)

import java.util.Scanner;

import static java.lang.Math.abs;
import static java.lang.Math.max;

public class Main {

    public static double nextDoubleBetween(double min, double max) {
        return (Math.random() * (max - min)) + min;
    }

    public static boolean inSquare(double x0, double y0, double r, double x, double y) {
        return max(abs(x - x0), abs(y - y0)) <= r;
    }

    public static boolean isBelong(double x, double y) {
        return inSquare(4.0, 2.0, 2.0, x, y)
                || inSquare(1.0, 1.0, 1.0, x, y)
                || inSquare(7.0, 1.0, 1.0, x, y)
                || inSquare(5.0, 5.0, 1.0, x, y)
                || inSquare(7.0, 5.0, 1.0, x, y)
                || inSquare(7.0, 7.0, 1.0, x, y);
    }

    public static void main(String[] args) {
        double width = 8, height = 8;
        double x0 = 0, y0 = 0;
        int count = 0;

        Scanner in = new Scanner(System.in);
        System.out.print("Количество итераций: ");
        long iterationsCount = in.nextLong();

        for (int i = 1; i <= iterationsCount; i++) {
            if (isBelong(nextDoubleBetween(x0, x0 + width), nextDoubleBetween(y0, y0 + height))) {
                count++;
            }
        }

        System.out.printf("Полученная площадь: %.2f \n", count*width*height/iterationsCount);
        in.close();
    }
}
