Практическое задание

Задача. Используя открытый ключ {3,22} и закрытый ключ {7,22} зашифровать и расшифровать слово МОСКВА

 

При решении задачи используем алфавит:

 

АБВГДЕЁЖЗИЙКЛМНОПРСТУФХЦЧШЩЬЫЪЭЮЯ

 

Заменим буквы алфавита цифрами, соответствующими их порядковому номеру в алфавите, исключив буквы Ё и Ъ

 

Решение.

 

Получим сообщение: МОСКВА → 13 15 18 11 3 1

 

Зашифруем сообщение с помощью открытого ключа {3, 22} для того, чтобы послать его абоненту А.

Используем преобразование: w' = wc (mod ri,) < ri.

w'1 =  (133)(mod22) = 19, 

w'2 = (153)(mod22) = 9, 

w'3 = (183)(mod22) = 2,

w'4 = (l13)(mod22) = 11,

w'5 = (33)(mod22) = 5,

w'6 = (l3)(mod22) = 1

 

Получили сообщение w' = (19, 9, 2, 11,5,1).

 

Абонент A расшифрует это сообщение (19, 9, 2, 11,5,1) с помощью закрытого ключа {7, 22}:

Используем преобразование: w² = (w¢)d (mod ri,). w²1 = (197)(mod 22) = 78125 (mod 22) = 13, 

w²2 = (97)(mod 22) = 15, 

w²3 = (27)(mod 22) =18, 

w²4 = (117)(mod 22) = 11,

w²5 = (57)(mod 22) = 3

w²6 = (17)(mod 22) = 1

 

Проверим результат из условия: w = w²

 

(13, 15, 18, 11,3,1) → МОСКВА

 

Для решения остальных задач была написана программа на C++

#include <iostream>
#include <valarray>

using namespace std;

const string dictionary[] = {"А", "Б", "В", "Г", "Д", "Е", "Ж", "З", "И", "Й", "К", "Л", "М", "Н", "О", "П", "Р", "С", "Т", "У", "Ф", "Х", "Ц", "Ч",
                             "Ш", "Щ", "Ы", "Ь", "Э", "Ю", "Я"};

// Получить код буквы
int getCode(const string& ch) {
    for(int i = 1; i <= size(dictionary); i++) {
        if (ch == dictionary[i - 1]) {
            return i;
        }
    }
    return -1;
}

// Полуить букву по коду
string getLetter(int index) {
    return dictionary[index - 1];
}

// Функция для нахождения НОД двух чисел
long long gcd(long long a, long long b) {
    while (b != 0) {
        long long temp = b;
        b = a % b;
        a = temp;
    }
    return a;
}

// Расширенный алгоритм Евклида для нахождения обратного элемента
long long modInverse(long long a, long long m) {
    long long g = gcd(a, m);
    if (g != 1) {
        cout << "Обратного элемента не существует." << endl;
        return -1;
    }
    else {
        // Используем расширенный алгоритм Евклида
        long long x1, y1, x2, y2, x, y, r1, r2, q, t;
        x2 = 1; x1 = 0; y2 = 0; y1 = 1;
        r2 = a; r1 = m;

        while(r1 > 0) {
            q = floor(r2 / r1);
            t = r2 % r1;
            x = x2 - q * x1;
            y = y2 - q * y1;

            r2 = r1; r1 = t;
            x2 = x1; x1 = x;
            y2 = y1; y1 = y;
        }

        if(x2 < 0)
            x2 += m;

        return x2;
    }
}

// Генерация открытого и закрытого ключей
void generateKeys(int p, int q, long long& e, long long& d) {
    long long phi = (p - 1) * (q - 1);

    // Выбираем случайное число e такое, что 1 < e < phi и НОД(e, phi) == 1
    for (e = 2; e < phi; e++) {
        if (gcd(e, phi) == 1) break;
    }

    // Находим обратный элемент d к e по модулю phi
    d = modInverse(e, phi);
}

int main() {

    setlocale(LC_ALL, "Russian");

    long long p = 2; // Первое простое число
    long long q = 11; // Второе простое число
    long long e, d;

    cout << "Выбранные простые числа: (" << p << ", " << q << ")" << endl;

    generateKeys(p, q, e, d);

    pair<long long, long long> publicKey = {e, p*q};
    pair<long long, long long> privateKey = {d, p*q};

    cout << "Открытый ключ: (" << publicKey.first << ", " << publicKey.second << ")" << endl;
    cout << "Закрытый ключ: (" << privateKey.first << ", " << privateKey.second << ")" << endl;

    string source[] = {"М", "О", "С", "К", "В", "А"};

    cout<<endl;
    cout << "Исходное сообщение: {";

    for (int i = 0; i < size(source) - 1; i++) {
        cout << source[i] << ", ";
    }
    cout << source[size(source) - 1] << "}" << endl;
    cout<<endl;

    long long code[size(source)];

    int index = 0;
    for(const auto & i : source) {
        code[index++] = static_cast<long long> (pow(getCode(i), publicKey.first)) % publicKey.second;
    }

    cout << "Зашифрованное сообщение: {";

    for (int i = 0; i < size(code) - 1; i++) {
        cout << code[i] << ", ";
    }
    cout << code[size(code) - 1] << "}" << endl;
    cout<<endl;

    cout << "Расшифрованое сообщение: {";

    for (int i = 0; i < size(code) - 1; i++) {
        cout << getLetter(static_cast<long long> (pow(code[i], privateKey.first)) % privateKey.second) << ", ";
    }
    cout << getLetter(static_cast<long long> (pow(code[size(code) - 1], privateKey.first)) % privateKey.second) << "}" << endl;
    cout<<endl;


    return 0;
}
